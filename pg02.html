const field01 = document.getElementById("field01")
const original = document.getElementById("original")

field01.addEventListener("input", amazing)      // create the function below then call it here

function amazing() {
    if (field01.value) {      // if field01 is not empty there is a value
        original.innerText = field01.value + " is fine"
    } else {
        original.innerText = "Please enter a name."     // field01 is empty - no value
    }
}
